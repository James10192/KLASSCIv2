# Details:
1. Implémenter les algorithmes de projection:
   ```php
   // Exemple de méthode de projection dans ComptabiliteService
   public function projeterCashFlow($mois = 3) {
       // Récupérer l'historique des 12 derniers mois
       $historique = Paiement::selectRaw('YEAR(date_paiement) as annee, MONTH(date_paiement) as mois, SUM(montant) as total')
           ->where('date_paiement', '>=', now()->subMonths(12))
           ->groupBy('annee', 'mois')
           ->orderBy('annee')
           ->orderBy('mois')
           ->get()
           ->toArray();
       
       // Calculer les tendances saisonnières
       $tendances = $this->calculerTendancesSaisonnieres($historique);
       
       // Projeter pour les prochains mois
       $projections = [];
       $dateActuelle = now();
       
       for ($i = 1; $i <= $mois; $i++) {
           $dateFuture = $dateActuelle->copy()->addMonths($i);
           $moisFutur = $dateFuture->month;
           
           // Appliquer tendance saisonnière + croissance moyenne
           $projection = $tendances[$moisFutur] * (1 + $this->tauxCroissanceMoyen());
           
           $projections[] = [
               'date' => $dateFuture->format('Y-m'),
               'montant_projete' => round($projection, 2),
               'intervalle_confiance' => $this->calculerIntervalleConfiance($projection)
           ];
       }
       
       return $projections;
   }
   ```
