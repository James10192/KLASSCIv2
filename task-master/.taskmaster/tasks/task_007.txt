2. Implémenter les jobs principaux:
   ```php
   // Exemple de job pour calcul KPIs
   class CalculerKPIsJob implements ShouldQueue {
       use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;
       
       protected $periode;
       
       public function __construct($periode = 'journalier') {
           $this->periode = $periode;
       }
       
       public function handle(ComptabiliteService $comptabiliteService) {
           $kpis = $comptabiliteService->calculerKPIs($this->periode);
           
           // Sauvegarder les KPIs calculés
           foreach($kpis as $nom => $valeur) {
               KPI::create([
                   'nom' => $nom,
                   'valeur' => $valeur,
                   'periode' => $this->periode,
                   'date_calcul' => now()
               ]);
           }
           
           // Déclencher événement pour mise à jour dashboard
           event(new KPIsCalcules($kpis));
       }
   }
   ```

4. Configurer les tâches planifiées:
   ```php
   // Dans App\Console\Kernel.php
   protected function schedule(Schedule $schedule) {
       $schedule->job(new CalculerKPIsJob('journalier'))->dailyAt('23:00');
       $schedule->job(new CalculerKPIsJob('hebdomadaire'))->weekly()->mondays()->at('01:00');
       $schedule->job(new CalculerKPIsJob('mensuel'))->monthlyOn(1, '02:00');
       
       $schedule->job(new SauvegardeDataJob())->dailyAt('03:00');
       
       // Vérifier les paiements en retard et planifier les relances
       $schedule->job(new PlanifierRelancesJob())->dailyAt('08:00');
   }
   ```
